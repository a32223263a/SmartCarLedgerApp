# Correcting the fueling records years
fueling_records = []

raw_lines = [line.strip() for line in fueling_text.strip().split('\n') if line.strip()]

# Hardcode the accurate years based on standard calendar matching (2025 for 09-12, 2026 for 03-06)
for line in raw_lines:
    if "2025년" in line or "2026년" in line:
        continue
    tokens = line.split()
    if len(tokens) >= 5:
        md = tokens[0] # "09/12"
        month = int(md.split('/')[0])
        day = int(md.split('/')[1])
        
        # If month is 9, 10, 11, 12 -> 2025
        # If month is 3, 4, 5, 6 -> 2026
        if month >= 9:
            year = 2025
        else:
            year = 2026
            
        date_str = f"{year}-{month:02d}-{day:02d}"
        day_of_week = tokens[1]
        price = int(tokens[2].replace(',', '').replace('₩', ''))
        liters = float(tokens[3].replace('L', ''))

# Convert DD-MM-YYYY to YYYY-MM-DD
def parse_driving_date(d_str):
    try:
        parts = str(d_str).strip().split('-')
        if len(parts) == 3:
            return f"{parts[2]}-{parts[1]}-{parts[0]}"
    except:
        pass
    return d_str

combined_df['주행일자_std'] = combined_df['주행일자'].apply(parse_driving_date)

import numpy as np

# Convert dates to datetime objects for accurate comparison
fueling_df['date_dt'] = pd.to_datetime(fueling_df['주유일자'])
driving_final['date_dt'] = pd.to_datetime(driving_final['주행일자'])

stats = []
for i in range(len(fueling_df)):
    current_fuel_date = fueling_df.loc[i, 'date_dt']
    liters = fueling_df.loc[i, '주유량(L)']
    price = fueling_df.loc[i, '주유금액(원)']
    
    if i == 0:
        # First fueling record: we don't have a previous point to measure distance since last fill
        stats.append({
            "주유일자": fueling_df.loc[i, '주유일자'],
            "주유량(L)": liters,
            "주유금액(원)": price,
            "구간주행거리(km)": 0.0,
            "구간연비(km/L)": 0.0,
            "누적연비(km/L)": 0.0
        })
    else:
        prev_fuel_date = fueling_df.loc[i-1, 'date_dt']
        
        # Distance driven between prev_fuel_date and current_fuel_date
        # Include driving records where prev_fuel_date < date <= current_fuel_date
        mask = (driving_final['date_dt'] > prev_fuel_date) & (driving_final['date_dt'] <= current_fuel_date)
        period_dist = driving_final.loc[mask, '주행거리(km)'].sum()
        
        # Fuel efficiency for this period = period_dist / liters filled at current date (Full-to-full assumption)
        period_efficiency = period_dist / liters if liters > 0 else 0.0
        
        stats.append({
            "주유일자": fueling_df.loc[i, '주유일자'],
            "주유량(L)": liters,
            "주유금액(원)": price,
            "구간주행거리(km)": period_dist,
            "구간연비(km/L)": round(period_efficiency, 2),
            "누적연비(km/L)": 0.0 # Will compute next
        })

stats_df = pd.DataFrame(stats)

# Compute cumulative efficiency: total distance / total liters (excluding first fill if we don't count its distance)
total_dist = 0.0
total_liters = 0.0
for i in range(len(stats_df)):
    if i > 0:
        total_dist += stats_df.loc[i, '구간주행거리(km)']
        total_liters += stats_df.loc[i, '주유량(L)']
        stats_df.loc[i, '누적연비(km/L)'] = round(total_dist / total_liters, 2) if total_liters > 0 else 0.0

print(stats_df)

import openpyxl
from openpyxl.styles import Font, PatternFill, Alignment, Border, Side
from openpyxl.utils import get_column_letter

# Create workbook
wb = openpyxl.Workbook()

# Setup sheets
ws_dashboard = wb.active
ws_dashboard.title = "연비 분석 대시보드"
ws_fueling = wb.create_sheet(title="주유 기록")
ws_driving = wb.create_sheet(title="통합 주행 기록")

# Common styles
font_family = "Arial"
header_font = Font(name=font_family, size=11, bold=True, color="FFFFFF")
data_font = Font(name=font_family, size=10)
title_font = Font(name=font_family, size=16, bold=True, color="1F4E79")

header_fill = PatternFill(start_color="2F4F4F", end_color="2F4F4F", fill_type="solid")
zebra_fill = PatternFill(start_color="F9FBFB", end_color="F9FBFB", fill_type="solid")
white_fill = PatternFill(start_color="FFFFFF", end_color="FFFFFF", fill_type="solid")

thin_border_side = Side(style='thin', color='D3D3D3')
thin_border = Border(left=thin_border_side, right=thin_border_side, top=thin_border_side, bottom=thin_border_side)

align_center = Alignment(horizontal="center", vertical="center")
align_right = Alignment(horizontal="right", vertical="center")
align_left = Alignment(horizontal="left", vertical="center")

# Function to write a dataframe into a sheet with formatting
def write_df_to_sheet(ws, df, title_text, col_formats=None):
    ws.views.sheetView[0].showGridLines = True
    
    # Title Block
    ws.append([])
    ws.append(["", title_text])
    ws.cell(row=2, column=2).font = title_font
    ws.row_dimensions[2].height = 25
    ws.append([])
    
    # Headers
    start_row = 4
    headers = df.columns.tolist()
    for col_idx, header in enumerate(headers, start=2):
        cell = ws.cell(row=start_row, column=col_idx, value=header)
        cell.font = header_font
        cell.fill = header_fill
        cell.alignment = align_center
        cell.border = thin_border
    ws.row_dimensions[start_row].height = 24
    
    # Data Rows
    for row_idx, row_data in enumerate(df.values, start=start_row + 1):
        ws.row_dimensions[row_idx].height = 20
        is_zebra = (row_idx % 2 == 0)
        current_fill = zebra_fill if is_zebra else white_fill
        
        for col_idx, val in enumerate(row_data, start=2):
            cell = ws.cell(row=row_idx, column=col_idx, value=val)
            cell.font = data_font
            cell.fill = current_fill
            cell.border = thin_border
            
            col_name = headers[col_idx - 2]
            # Formats and alignments
            if col_formats and col_name in col_formats:
                fmt, alignment = col_formats[col_name]
                cell.number_format = fmt
                cell.alignment = alignment
            else:
                if isinstance(val, (int, float)):
                    cell.number_format = "#,##0"
                    cell.alignment = align_right
                else:
                    cell.alignment = align_center

    # Adjust Column Widths
    for col in ws.columns:
        max_len = 0
        col_letter = get_column_letter(col[0].column)
        if col[0].column == 1:
            ws.column_dimensions[col_letter].width = 3
            continue
        for cell in col:
            if cell.row >= start_row and cell.value:
                max_len = max(max_len, len(str(cell.value)))
        ws.column_dimensions[col_letter].width = max(max_len + 4, 12)

# Define column formatting for each sheet
dashboard_formats = {
    "주유일자": ("yyyy-mm-dd", align_center),
    "주유량(L)": ("#,##0.00", align_right),
    "주유금액(원)": ("#,##0", align_right),
    "구간주행거리(km)": ("#,##0.0", align_right),
    "구간연비(km/L)": ("#,##0.00", align_right),
    "누적연비(km/L)": ("#,##0.00", align_right),
}

fueling_formats = {
    "주유일자": ("yyyy-mm-dd", align_center),
    "요일": ("@", align_center),
    "주유금액(원)": ("#,##0", align_right),
    "주유량(L)": ("#,##0.00", align_right),
    "단가(원/L)": ("#,##0", align_right),
}

driving_formats = {
    "주행일자": ("yyyy-mm-dd", align_center),
    "시동시각": ("hh:mm", align_center),
    "종료시각": ("hh:mm", align_center),
    "주행거리(km)": ("#,##0.0", align_right),
    "주행시간(분)": ("#,##0", align_right),

# Write sheets
write_df_to_sheet(ws_dashboard, stats_df, "구간별 및 누적 연비 분석 리포트", dashboard_formats)
write_df_to_sheet(ws_fueling, fueling_df.drop(columns=['date_dt']), "주유 기록 내역 데이터", fueling_formats)
write_df_to_sheet(ws_driving, driving_final.drop(columns=['date_dt']), "월별 통합 주행 기록 데이터", driving_formats)

# Freeze panes for convenience
ws_dashboard.freeze_panes = "B5"
ws_fueling.freeze_panes = "B5"
ws_driving.freeze_panes = "B5"

## 1. SOLID 원칙을 만족하는 앱 시스템 아키텍처 설계

매달 새로운 주행 기록 파일(`driving_YYYYMM.xlsx`)이 추가되고 주유 기록 또한 지속해서 축적되므로, 시스템은 확장에는 열려 있고 수정에는 닫혀 있는 OCP(개방-폐쇄 원칙)와 고수준 모듈이 저수준 모듈의 구현에 의존하지 않는 DIP(의존 역전 원칙)를 반드시 만족해야 합니다.

### 📊 아키텍처 구조도 (컴포넌트 설계)

1. **도메인 엔티티 (Domain Entities)**
* `DrivingRecord`: 주행일자, 시동/종료시각, 주행거리, 주행시간 데이터 구조.
* `FuelingRecord`: 주유일자, 요일, 주유금액, 주유량, 단가 데이터 구조.
* `EfficiencyStats`: 특정 구간의 주행거리, 구간 연비, 누적 연비를 담는 구조.


2. **추상화 레이어 (Interfaces - OCP & DIP의 핵심)**
* `DrivingLogParser` (Interface): 새로운 주행기록 파일을 읽어 `List<DrivingRecord>`로 변환하는 규격 정의.
* `FuelingRepository` (Interface): 주유기록을 저장, 로드, 엑셀 내보내기/가져오기 처리를 정의하는 인터페이스.
* `EfficiencyCalculator` (Interface): 연비 계산 알고리즘 규격 정의.


3. **구현 레이어 (Concrete Classes)**
* `ExcelDrivingLogParser`: openpyxl 또는 pandas 라이브러리를 사용해 엑셀 형식의 주행기록을 파싱하는 클래스.
* `ExcelFuelingRepository`: 주유기록을 파일 시스템(Excel)에 저장하고 필요시 다시 빼오기(Export)를 수행하는 클래스.

## 2. 핵심 비즈니스 로직: 연비 계산 알고리즘

주유소에서 주유할 때 항상 **'가득(Full-to-Full)'** 채운다고 가정할 때 가장 정확한 연비 계산법이 적용됩니다.

1. **구간 주행거리 계산**:
* 주유 기록 $N-1$번의 '주유일자' 다음 날부터 주유 기록 $N$번의 '주유일자'까지 발생한 **통합 주행 기록의 모든 `주행거리(km)`의 합산**을 구합니다.


2. **구간 연비**:
* $\text{구간 연비} = \frac{\text{구간 주행거리}(km)}{\text{주유 기록 } N\text{번의 주유량}(L)}$
* *이유*: 주유소에 들어가 가득 채운 양($L$)은 직전 주유 이후부터 지금까지 달린 거리만큼 소모된 연료량과 정확히 일치하기 때문입니다.


3. **누적 연비**:
* $\text{누적 연비} = \frac{\sum(\text{전체 구간 주행거리})}{\sum(\text{첫 주유를 제외한 전체 주유량})}$

## 4. 앱 개발을 위한 단계적 구현 로드맵

* **1단계: 데이터 파싱 엔진 구축 (Core Backend)**
* 추상 인터페이스 `DrivingLogParser`를 선언하고 구현체에서 `openpyxl` 라이브러리를 사용해 업로드된 엑셀 파일들을 파싱하여 날짜별 정렬 데이터로 가공합니다.


* **2단계: 저장소 및 입출력 모듈 구현 (Data IO Layer)**
* `FuelingRepository`를 통해 사용자가 화면에서 입력한 주유 내역을 로컬 SQLite 데이터베이스 또는 지정된 엑셀 파일(`fueling_database.xlsx`)에 자동으로 삽입(Row Append)하는 모듈을 만듭니다. 반대로 엑셀 추출 요청 시 이 스트림을 파일로 빌드하여 다운로드할 수 있도록 합니다.


* **3단계: 연비 연산 엔진 연동 (Business Logic)**
* 위의 2번 섹션에 기재된 **Full-to-Full 연비 알고리즘**을 구현하여 데이터가 추가될 때마다 대시보드 스탯을 갱신하는 코드를 결합합니다.


* **4단계: Figma UI 기반 프론트엔드 연결 (UI Integration)**
* Figma로 도출된 디자인 컴포넌트 구조를 바탕으로 Flutter, React Native, 또는 웹 UI(React/Vue) 화면을 생성하고 백엔드 비즈니스 로직과 API 인터페이스를 바인딩하여 앱을 완성합니다.
